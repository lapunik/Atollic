#include <stdio.h>

#include "ansi.h"

int AnsiSendValue(char cValue)
{
	if (cValue==0)
	{
		putchar('0');
		return 1;
	}

	if (cValue>=100)			// nejvyssi rad ?
	{
		putchar(cValue/100 + '0');
		cValue%=100;			// a zahod ho

		putchar(cValue/10 + '0');	// desitky

	}
	else
	{
		if (cValue>=10)
			putchar(cValue/10 + '0');	// desitky
	}

	cValue%=10;					// a jednotky
	putchar(cValue + '0');

	return 1;					// provedeno OK
}

int AnsiSetGrMode(char cTextAttr, char cFore, char cBack)
{
	if ((cTextAttr==2)||(cTextAttr==3)||(cTextAttr>8))
		return 0;				// zakazane hodnoty = false

	if (cFore<30)				// mozno zadat jako 0-7
		cFore+=30;

	if ((cFore<30)||(cFore>37))
		return 0;				// zakazane hodnoty = false

	if (cBack<40)				// mozno zadat jako 0-7
		cBack+=40;

	if ((cBack<40)||(cBack>47))
		return 0;				// zakazane hodnoty = false

	putchar('\033');
	putchar('[');

	AnsiSendValue(cTextAttr);
	putchar(';');

	AnsiSendValue(cFore);
	putchar(';');

	AnsiSendValue(cBack);
	putchar('m');
	return 1;
}

int AnsiCurMove(char cDir, char cNum)
{
	switch(cDir)
	{
		case	DIR_UP:
			return AnsiCurUp(cNum);
		case	DIR_DOWN:
			return AnsiCurDown(cNum);
		case	DIR_RIGHT:
			return AnsiCurRight(cNum);
		case	DIR_LEFT:
			return AnsiCurLeft(cNum);
		default:
			return 0;				// false. spatna hodnota
	}
}

int AnsiCurUp(char cNum)
{
	putchar('\033');
	putchar('[');

	AnsiSendValue(cNum);

	putchar('A');
	return 1;
}

int AnsiCurDown(char cNum)
{
	putchar('\033');
	putchar('[');

	AnsiSendValue(cNum);

	putchar('B');
	return 1;
}

int AnsiCurRight(char cNum)
{
	putchar('\033');
	putchar('[');

	AnsiSendValue(cNum);

	putchar('C');
	return 1;
}

int AnsiCurLeft(char cNum)
{
	putchar('\033');
	putchar('[');

	AnsiSendValue(cNum);

	putchar('D');
	return 1;
}


int AnsiCurPos(char cColumn, char cLine)
{
	if ((cLine>50)||(cLine<0))
		return 0;

	if ((cColumn>80)||(cColumn<0))
		return 0;

	putchar('\033');
	putchar('[');

	AnsiSendValue(cLine);
	putchar(';');

	AnsiSendValue(cColumn);

	putchar('H');
	return 1;
}


int AnsiSetMode(char cMode)
{
	if ((cMode>19)||(cMode<0))
		return 0;

	putchar('\033');
	putchar('[');
	putchar('=');

	AnsiSendValue(cMode);

	putchar('h');
	return 1;
}

int AnsiResetMode(char cMode)
{
	if ((cMode>19)||(cMode<0))
		return 0;

	putchar('\033');
	putchar('[');
	putchar('=');

	AnsiSendValue(cMode);

	putchar('l');				// male L !!!
	return 1;
}

int AnsiPuts(char *cPtr)
{
	for (;*cPtr;cPtr++)
		putchar(*cPtr);

	return 1;
}
